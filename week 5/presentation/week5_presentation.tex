\begin{frame}[fragile]{Fortran BLAS Signatures}
  All kernels follow the \textbf{Fortran BLAS ABI}: trailing underscore, all arguments by pointer.

  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{Wrong (CBLAS):}
      \begin{lstlisting}
// CBLAS - not compatible with
// Fortran calling convention
void cblas_sgemv(
    CBLAS_ORDER order,
    CBLAS_TRANSPOSE TransA,
    int M, int N,
    float alpha,
    const float *A, int lda, ...);
      \end{lstlisting}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{Correct (Fortran BLAS):}
      \begin{lstlisting}
// Fortran ABI - drop-in for
// OpenBLAS in linked libraries
void sgemv_(
    char *trans,
    int *m, int *n,
    float *alpha,
    const float *a, int *lda,
    const float *x, int *incx,
    float *beta,
    float *y, int *incy);
      \end{lstlisting}
    \end{column}
  \end{columns}
  \vspace{0.3em}
  \begin{exampleblock}{Why it matters}
    BLAS 2 \& 3 functions will be linked into \texttt{libblas.a} replacing OpenBLAS in benchmarks. Fortran ABI is required for compatibility with \texttt{-lblas} and \texttt{-lopenblas} linker flags.
  \end{exampleblock}
\end{frame}

\section{Test Design}

\begin{frame}{Test Design: What We Test and Why}
  \begin{columns}[T]
    \begin{column}{0.5\textwidth}
      \textbf{Why not test random cases?}\\
      Random data misses the real failure modes. We test \textbf{hot spots}:
      \begin{itemize}
        \item Sizes: $n \in \{32, 128, 512, 1024\}$ - L1/L2/L3 cache boundaries
        \item alpha $\in \{0, 1, -1, 0.5\}$ - zero skips, unit avoids multiply
        \item beta $\in \{0, 1, -1\}$ - beta=0 is dominant in blocked GEMM
        \item Upper \textbf{and} Lower triangular for ssymv, ssyr, strmv, strsv
        \item Unit \textbf{and} Non-unit diagonal for strmv, strsv
        \item NoTrans \textbf{and} Trans for sgemv, strmv, strsv
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{Edge cases that found bugs:}
      \begin{itemize}
        \item \texttt{strmv Upper Trans NonUnit} $n=128$:\\
              exposed transpose access bug
        \item \texttt{strsv Lower NonTrans} $n=512$:\\
              ill-conditioned matrix (uniform diagonal $\Rightarrow$ condition $\sim 3^n$)
        \item \texttt{ssyrk} $n=256$, $k=128$:\\
              wrong sgemm\_ex product (max error 0.4)
      \end{itemize}
      \vspace{0.4em}
      \begin{alertblock}{Test matrix conditioning}
        \scriptsize
        Triangular tests: diagonal $= 2 + \text{rand}()$,\\
        off-diagonal $= 0.01 \cdot \text{rand}()$\\
        $\Rightarrow$ condition number $\approx O(1)$, not $O(3^n)$
      \end{alertblock}
    \end{column}
  \end{columns}
  \vspace{0.2em}
  \centering\footnotesize Total: \textbf{118/118} BLAS~2+3 tests pass \quad|\quad \textbf{1287/1287} BLAS~1 tests pass
\end{frame}

\section{TASK2: True 3D Parallelism and Reduction Tree}
% ==================================================================

\begin{frame}{TASK2: Making It Actually 3D}
  \begin{columns}[T]
    \begin{column}{0.5\textwidth}
      \textbf{Before: "3D" in name only}
      \begin{itemize}
        \item K dimension split into $r$ tiles
        \item Each tile: separate partial $C$ buffer
        \item \alert{Reduction: sequential loop over $r$ buffers} with implicit barrier at end of \texttt{parallel for}
        \item Not truly 3D: K-tiles processed by same thread team, not independent tasks
      \end{itemize}
      \vspace{0.3em}
      \textbf{After: true 3D with reduction tree}
      \begin{itemize}
        \item $M \times N \times K$ decomposition: all three dimensions assigned to distinct thread groups
        \item K-tiles computed concurrently by independent task groups
        \item Reduction: \textbf{binary tree} - $r$ buffers summed in $\log_2 r$ parallel steps
        \item No global barrier until final write
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{tikzpicture}[scale=0.85,
        buf/.style={draw,fill=hpcblue!20,rounded corners,minimum width=1.2cm,minimum height=0.5cm,font=\scriptsize},
        arr/.style={-{Latex},thick}
      ]
        % Level 0: 4 partial buffers
        \node[buf] (p0) at (0,0)   {$P_0$};
        \node[buf] (p1) at (1.5,0) {$P_1$};
        \node[buf] (p2) at (3,0)   {$P_2$};
        \node[buf] (p3) at (4.5,0) {$P_3$};
        % Level 1: two sums
        \node[buf,fill=hpcgreen!20] (s01) at (0.75,1.2) {$P_0{+}P_1$};
        \node[buf,fill=hpcgreen!20] (s23) at (3.75,1.2) {$P_2{+}P_3$};
        % Level 2: final
        \node[buf,fill=hpcred!20] (fin) at (2.25,2.4) {$C = \sum P_i$};
        \draw[arr] (p0) -- (s01);
        \draw[arr] (p1) -- (s01);
        \draw[arr] (p2) -- (s23);
        \draw[arr] (p3) -- (s23);
        \draw[arr] (s01) -- (fin);
        \draw[arr] (s23) -- (fin);
        \node[font=\tiny,hpcgray] at (2.25,-0.5) {$r=4$: 2 tree levels vs 3 sequential steps};

\begin{frame}[fragile]{FMA in Reduction Step: Investigation}
  \textbf{Question from assignment}: Why do we use FMA in the reduction step? Is there a more efficient way?

  \begin{columns}[T]
    \begin{column}{0.5\textwidth}
      \textbf{Original reduction} (general fmadd):
      \begin{lstlisting}
__m256 vb = _mm256_set1_ps(beta);
// C[j] = beta*C[j] + P[j]
_mm256_storeu_ps(C+j,
  _mm256_fmadd_ps(vb,
    _mm256_loadu_ps(C+j),
    _mm256_loadu_ps(P+j)));
      \end{lstlisting}
      \alert{Problem}: reads $C$ even when $\beta=0$ (wasteful). FMA = read + multiply + add = 3 ops, but when $\beta \in \{0,1\}$ the multiply is trivial.
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{Optimized: beta fast-paths}
      \begin{lstlisting}
if (beta == 0.0f)
  // pure copy - no C read
  storeu(C+j, loadu(P+j));
else if (beta == 1.0f)
  // add only - no multiply
  storeu(C+j, add_ps(
    loadu(C+j), loadu(P+j)));
else
  storeu(C+j, fmadd(vb,
    loadu(C+j), loadu(P+j)));
      \end{lstlisting}
      \begin{exampleblock}{Result ($n=1024$, 20T, $\beta=0$)}
        \scriptsize 751 GF/s $\to$ \textbf{795 GF/s} (+5.9\%)\\
        $\beta=0$ is dominant in blocked GEMM (first tile)
      \end{exampleblock}
    \end{column}
  \end{columns}
\end{frame}

        512  & 537   & 25.1  & 10.2 & 217  \\
        1024 & 752   & 64.9  & 29.6 & 481  \\
        2048 & 693   & 65.5  & 85.4 & 533  \\
        \bottomrule
      \end{tabular}\\[0.2em]
      \textit{\scriptsize GF/s, 20 threads}
      \end{center}

      \vspace{0.5em}
      \alert{\texttt{taskloop} is 10--12$\times$ slower than \texttt{parallel for}} for medium $n$ due to task graph scheduling overhead ($O(\text{tile\_count}^2)$ dependency tracking).
    \end{column}
    \begin{column}{0.43\textwidth}
      \begin{block}{Why tasks can \emph{theoretically} win}
        \begin{itemize}
          \item \texttt{parallel for} has implicit barrier at loop end
          \item Tasks with dependencies: threads move to next tile without waiting
          \item For very large $n$ and many threads: less idle time
        \end{itemize}
      \end{block}
      \vspace{0.3em}
      \begin{alertblock}{Why tasks lose in practice}
        \begin{itemize}
          \item Fine-grained tiles: task creation overhead $\gg$ work
          \item OpenMP runtime spends more time scheduling than computing
          \item \texttt{parallel for collapse(2) schedule(dynamic)} wins
        \end{itemize}
      \end{alertblock}
    \end{column}
  \end{columns}
\end{frame}

\section{BLAS 2 Kernels}

\begin{frame}[fragile]{sgemv --- Matrix-Vector Multiply}
  $\mathbf{y} = \alpha A \mathbf{x} + \beta \mathbf{y}$ \quad or \quad
  $\mathbf{y} = \alpha A^T \mathbf{x} + \beta \mathbf{y}$

  \begin{columns}[T]
    \begin{column}{0.5\textwidth}
      \textbf{NoTrans (row access)} --- vectorizable\\[0.3em]
      Dual 8-wide accumulators hide 4-cycle FMA latency:
      \begin{lstlisting}
__m256 acc0 = _mm256_setzero_ps();
__m256 acc1 = _mm256_setzero_ps();
for (; k <= n-16; k += 16) {
  acc0 = _mm256_fmadd_ps(
    _mm256_loadu_ps(Arow+k),
    _mm256_loadu_ps(x+k), acc0);
  acc1 = _mm256_fmadd_ps(
    _mm256_loadu_ps(Arow+k+8),
    _mm256_loadu_ps(x+k+8), acc1);
}
      \end{lstlisting}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{Trans (column access)} --- cache hostile\\[0.3em]
      $A[k, j]$ stride = $\text{lda}\cdot 4$ bytes between consecutive elements.

      \medskip
      \alert{For $n=1024$: stride = 4~KB $=$ L1 cache line granularity $\Rightarrow$ aliasing}

      \medskip
      \textbf{What OpenBLAS does}: load a $k_c \times n_r$ panel once, accumulate $n_r$ outputs simultaneously (amortizes column loads).

      \medskip
      \textbf{Our approach}: parallel over $k$ (rows), scalar inner loop. Column access bottleneck remains.
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}[fragile]{ssyr2 --- Fused Dual-FMA: Our Best Result}
  $A = \alpha(\mathbf{x}\mathbf{y}^T + \mathbf{y}\mathbf{x}^T) + A$

  \textbf{Key insight}: both updates touch the same row of $A$ - fuse into \textbf{one memory pass}:
  \begin{lstlisting}
void ssyr2_(char *uplo, int *n, float *alpha, float *x, int *incx,
            float *y, int *incy, float *a, int *lda) {
    ...
    __m256 vxi = _mm256_set1_ps(*alpha * x[i]);
    __m256 vyi = _mm256_set1_ps(*alpha * y[i]);
    for (; j <= i-8; j += 8) {
        __m256 av = _mm256_loadu_ps(Arow+j);
        av = _mm256_fmadd_ps(vxi, _mm256_loadu_ps(y+j), av); // A += alpha*x[i]*y[j]
        av = _mm256_fmadd_ps(vyi, _mm256_loadu_ps(x+j), av); // A += alpha*y[i]*x[j]
        _mm256_storeu_ps(Arow+j, av);
    }
}
  \end{lstlisting}
  \begin{columns}
    \begin{column}{0.48\textwidth}
      \begin{exampleblock}{Memory traffic analysis}
        \footnotesize
        Naive (2 separate ssyr calls): $4 \times n^2/2$ floats\\
        Fused: $2 \times n^2/2$ floats \;\textbf{(halved for $A$)}
      \end{exampleblock}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{ssyr2 Fortran entry point}
        \texttt{void ssyr2\_(char*, int*, float*, ...)}
        
        Wrapper calls C implementation with dereferenced args.
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}[fragile]{ssymv --- Attempt to Eliminate Branch}
  $\mathbf{y} = \alpha A \mathbf{x} + \beta \mathbf{y}$, symmetric $A$ (one triangle stored)

  \begin{columns}[T]
    \begin{column}{0.5\textwidth}
      \textbf{Iteration 1 --- branched access}
      \begin{lstlisting}
float aij = (i<=j) ? a[i*lda+j]
                   : a[j*lda+i];
y[i] += aij * x[j];
      \end{lstlisting}
      Branch prevents auto-vectorization $\Rightarrow$ 2~GB/s.

      \medskip
      \textbf{Iteration 2 --- split triangle}\\
      Stored half: contiguous row $\Rightarrow$ \textbf{AVX2}\\
      Reflected half: column stride $\Rightarrow$ \textbf{scalar}\\[0.3em]
      \alert{Result}: 1.6--2.8 GB/s vs OpenBLAS 22 GB/s\\
      AVX2 gain on stored half cancelled by reflected half
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{Root cause}

      Reflected half: $A[k \cdot \text{lda} + i]$, $k$ iterating\\
      $\Rightarrow$ stride $= 4n$ bytes per load\\
      $\Rightarrow$ one cache line per element\\
      $\Rightarrow$ $n^2/2$ L2 misses for $n=1024$

      \medskip
      \textbf{What is needed}: panel accumulation\\
      Load each \emph{column} once, scatter into 4--8 output rows simultaneously.

      \medskip
      OpenBLAS: $\sim$800 lines assembly per architecture.

      \vspace{0.3em}
      \alert{Gap accepted and documented.}\\
      ssymv is excluded from AVX2 optimization list in README.
    \end{column}
  \end{columns}
\end{frame}

\section{BLAS 1 --- Non-Unit Stride}

\begin{frame}[fragile]{BLAS 1 --- AVX2 Gather for inc $>$ 1}
  Assignment: \textit{``if incX or incY is not 1, use a more efficient operation.''}

  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \textbf{Gather index helper (written once, reused everywhere)}
      \begin{lstlisting}
static inline __m256i make_gather_idx(int inc) {
    return _mm256_mullo_epi32(
        _mm256_set_epi32(7,6,5,4,3,2,1,0),
        _mm256_set1_epi32(inc));
}

// sdot, snrm2, sasum, saxpy with incx > 1:
__m256i idx = make_gather_idx(incx);
__m256 xv = _mm256_i32gather_ps(
               x + i*incx, idx, 4);
      \end{lstlisting}
      Capped at inc $\leq 1024$ to avoid degenerate patterns.
    \end{column}
    \begin{column}{0.46\textwidth}
      \textbf{Performance (sdot, inc=2, $n=65536$)}
      \begin{itemize}
        \item Scalar: 18.2 GB/s
        \item AVX2 gather: \textbf{28.4 GB/s} ($+56\%$)
        \item OpenBLAS inc=2: 22.7 GB/s
        \item Unit stride AVX2: 42.1 GB/s (reference)
      \end{itemize}
      \vspace{0.3em}
      Gather: $\sim$1 elem/cycle vs loadu: 8 elem/cycle\\
      Still beats scalar due to eliminated pointer arithmetic overhead.

      \vspace{0.3em}
      Applied to: \textbf{saxpy, sdot, snrm2, sasum}
    \end{column}
  \end{columns}
\end{frame}

\section{Week 6 (Bonus): BLAS 3}
% ==================================================================

\begin{frame}[fragile]{Week 6: ssyrk, strmm, strsm}
  \begin{columns}[T]
    \begin{column}{0.5\textwidth}
      \textbf{ssyrk}: $C = \alpha A A^T + \beta C$

      \medskip
      \textit{Failed approach}: delegate to sgemm\_ex\\
      \alert{Bug}: sgemm\_ex has no transposed-B support. Passes $A[j\text{-tile},:]$ as $K \times jb$, but it is $jb \times k$. Max error: 0.4.

      \medskip
      \textit{Fix}: explicit tiled AVX2 dot products:
      \begin{lstlisting}
// C[ii,jj] = alpha * dot(A[ii,:], A[jj,:])
const float *Ai = a + ii * lda; // contiguous
const float *Aj = a + jj * lda; // contiguous
__m256 acc = _mm256_setzero_ps();
for (kk=0; kk<=k-8; kk+=8)
  acc = _mm256_fmadd_ps(
    _mm256_loadu_ps(Ai+kk),
    _mm256_loadu_ps(Aj+kk), acc);
      \end{lstlisting}
      Both rows accessed contiguously. Tiled $128\times128$ for L2 reuse.
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{strmm}: $B = \alpha \,\mathrm{op}(A) \cdot B$

      Implemented via temporary copy of $B$ + explicit triangular matrix-vector pass. All 8 variants (Side $\times$ Uplo $\times$ Trans $\times$ Diag). Sequential, correctness focus.

      \vspace{0.8em}
      \textbf{strsm}: $\mathrm{op}(A) \cdot X = \alpha B$

      Full back-/forward-substitution for all 8 variants:
      \begin{itemize}
        \item Left Upper NoTrans: backward substitution
        \item Left Lower NoTrans: forward substitution
        \item Left Upper Trans, Left Lower Trans
        \item Right-side variants (X outer loop over rows)
      \end{itemize}

      \begin{exampleblock}{All BLAS 3 correctness tests pass}
        \scriptsize ssyrk (5 cases), strmm (included in 118 total)
      \end{exampleblock}
    \end{column}
  \end{columns}
\end{frame}
